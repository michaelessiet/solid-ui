## Usage

```tsx
import { RadioGroup, RadioGroupItem } from "~/components/ui/radio-group"
```

```tsx
<RadioGroup>
  <For each={["Apple", "Orange", "Watermelon"]}>
    {(fruit) => (
      <RadioGroupItem value={fruit}>
        <Label>{fruit}</Label>
      </RadioGroupItem>
    )}
  </For>
</RadioGroup>
```
